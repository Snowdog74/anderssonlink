CREATE PROCEDURE InsertOrder 
	-- Add the parameters for the stored procedure here
	@ArticleNo int, 
	@Description nvarchar(50),
	@OrderDate date,
	@DeliveryDate date,
	@Quantity int,
	@PiecePrice int,
	@Customer nvarchar(50),
	@CustomerNo int,
	@CustomerOrderNo int,
	@Info text,
	@Currency nchar(3),
	@Recipient nvarchar(50)
AS
BEGIN
	-- SET NOCOUNT ON added to prevent extra result sets from
	-- interfering with SELECT statements.
	SET NOCOUNT ON;

    -- Insert statements for procedure here
    INSERT INTO dbo.Orders 
    (
		ArticleNo,
		Description,
		OrderDate,
		DeliveryDate,
		Quantity,
		PiecePrice,
		Customer,
		CustomerNo,
		CustomerOrderNo,
		Info,
		Currency,
		Recipient) 
    VALUES 
    (
		@ArticleNo,
		@Description,
		@OrderDate,
		@DeliveryDate,
		@Quantity,
		@PiecePrice,
		@Customer,
		@CustomerNo,
		@CustomerOrderNo,
		@Info,
		@Currency,
		@Recipient);
	RETURN @@ERROR
END
GO
